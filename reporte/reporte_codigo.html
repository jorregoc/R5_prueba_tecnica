Consistencia - formato str

Se crea una funcin para evaluar si los campos que reciben strings presentan el mismo formato. Es decir, si el registro est en minsculas, maysculas o mayscula inicial. Se omiten los campos Id ya que contienen llaves nicas, por lo cual pueden mezclar letras y nmeros.

def verificar_consistencia(df, columnas):
    '''
    Evala si todos los valores de una columna presentan el mismo formato en cuanto a
    minsculas, maysculas y maysculas iniciales.
    param:: df: Dataframe a evaluar
            columnas: columnas del dataframe a evaluar (lista)
    '''
    columnas_inconsistentes = [] #para almacenar los nombres de las columnas que no cumplen con la condicin dada.
    
    for columna in columnas:
        valores = df[columna]
        
        # Verificar si todos los valores estn en minsculas, maysculas o en mayscula inicial
        if valores.str.islower().all() or valores.str.isupper().all() or valores.str.istitle().all():
            continue  # Todos los valores tienen la misma consistencia en esta columna
        
        columnas_inconsistentes.append(columna)
    
    return columnas_inconsistentes

def verificar_consistencia_decimales(df, columnas):
    '''
     Evala si todas los registros en las columnas indicadas
     contienen la misma cantidad de decimales.
     param:: df: Dataframe a evaluar
             columnas: columnas del dataframe a evaluar (lista)
    '''
    columnas_inconsistentes = []

    for columna in columnas:
        valores = df[columna]

        # Convertir los valores a cadenas y obtener la cantidad de decimales en cada valor
        decimales = valores.apply(lambda x: len(str(x).split('.')[-1]) if isinstance(x, float) else 0)

        if decimales.nunique() > 1:

    return columnas_inconsistentes
